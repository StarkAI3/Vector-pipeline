# ============================================================================

# This system now supports multiple vector databases!

# - Pinecone (cloud/serverless)

# ============================================================================
# VECTOR DATABASE CONFIGURATION (MODULAR)
# ============================================================================
# Choose your vector database: pinecone, qdrant, weaviate, chroma, milvus
VECTOR_DB_TYPE=pinecone

# Deployment type: "cloud" or "local"
VECTOR_DB_DEPLOYMENT=cloud

# Generic settings (apply to all databases)
VECTOR_DB_INDEX_NAME=dma-bot-index
VECTOR_DB_DIMENSION=768
VECTOR_DB_METRIC=cosine
VECTOR_DB_BATCH_SIZE=100
VECTOR_DB_NAMESPACE=

# ============================================================================
# OPTION 1: PINECONE CONFIGURATION (Cloud)
# ============================================================================
# Use these when VECTOR_DB_TYPE=pinecone
VECTOR_DB_API_KEY=your_pinecone_api_key_here
VECTOR_DB_CLOUD_PROVIDER=aws
VECTOR_DB_REGION=us-east-1

# Legacy Pinecone variables (for backward compatibility)
# If VECTOR_DB_API_KEY is empty, these will be used
PINECONE_API_KEY=your_pinecone_api_key_here
PINECONE_ENVIRONMENT=us-east-1
PINECONE_INDEX_NAME=dma-bot-index

# ============================================================================
# OPTION 2: QDRANT CONFIGURATION
# ============================================================================
# Use these when VECTOR_DB_TYPE=qdrant

# For Qdrant Cloud:
VECTOR_DB_TYPE=qdrant
VECTOR_DB_DEPLOYMENT=cloud
VECTOR_DB_API_KEY=your_qdrant_cloud_api_key
VECTOR_DB_HOST=your-cluster.qdrant.io
VECTOR_DB_PORT=6333
VECTOR_DB_INDEX_NAME=my_collection
VECTOR_DB_DIMENSION=768
VECTOR_DB_METRIC=cosine

# For Local Qdrant:
# VECTOR_DB_TYPE=qdrant
# VECTOR_DB_DEPLOYMENT=local
# VECTOR_DB_HOST=localhost
# VECTOR_DB_PORT=6333
# VECTOR_DB_GRPC_PORT=6334
